47 auto addSamples =

197 Token detectSymbolFromFftResults(const Fft::FftResults& results) const {
198 std::vector<Fft::FftResults::FreqAmp> sorted_peaks = results.getNormalizedPeaks(
199 NUM_PEAK_FREQS_TO_CHECK, MIN_AMPLITUDE_CUT_OFF_PARAMETER, MIN_FREQ_CUT_OFF,
200 MAX_FREQ_CUT_OFF);

202 // std::cout << "f,A: ";

204 // std::cout << "[" << peak.frequency << ", " << peak.amplitude << "]";
205 // }
206 // std::cout << std::endl;

208 // size_t num_0_peaks_of_interest = 0;
209 // size_t num_1_peaks_of_interest = 0;
210 double sym_0_poi_power = 0;
211 double sym_1_poi_power = 0;
212 double sym_2_poi_power = 0;
213 double sym_3_poi_power = 0;
214
215 // Start at a non-zero for the sake of a sensible SNR value later on.
216 // SNR would be better calculated by taking the integral of the whole FFT
217 // and then subtracting the peaks of interest.
218 double power_outside_poi = 0.2;
219
220 for (const auto& res : sorted_peaks) {
221 const double peak_to_check = res.frequency;
222 bool peak_used_by_symbol = false;
223
224 for (double peak_of_interest : SYM_A_00_PEAKS_OF_INTEREST) {
225 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
226 sym_0_poi_power += res.amplitude;
227 peak_used_by_symbol = true;
228 }
229 }
230 for (double peak_of_interest : SYM_D_01_PEAKS_OF_INTEREST) {
231 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
232 sym_1_poi_power += res.amplitude;
233 peak_used_by_symbol = true;
234 }
235 }
236 for (double peak_of_interest : SYM_E_10_PEAKS_OF_INTEREST) {
237 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
238 sym_2_poi_power += res.amplitude;
239 peak_used_by_symbol = true;
240 }
241 }
242 for (double peak_of_interest : SYM_G_11_PEAKS_OF_INTEREST) {
243 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
244 sym_3_poi_power += res.amplitude;
245 peak_used_by_symbol = true;
246 }
247 }
248
249 if (!peak_used_by_symbol) {
250 power_outside_poi += res.amplitude;
251 }
252 }
253 sym_0_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
254 sym_1_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
255 sym_2_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
256 sym_3_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
257 power_outside_poi /= NUM_PEAK_FREQS_TO_CHECK;
258
259 const double average_working_power =
260 (sym_0_poi_power + sym_1_poi_power + sym_2_poi_power + sym_3_poi_power) / 4.0;
261
262 const double signal_to_noise_ratio = average_working_power / power_outside_poi;
263
264 if (average_working_power < MIN_AMPLITUDE_CUT_OFF_PARAMETER) {
265 return Token::UNKNOWN;
266 }

268 std::vector<std::pair<Token, double>> symbol_powers{

270 {Token::SYMBOL_01, sym_1_poi_power},
271 {Token::SYMBOL_10, sym_2_poi_power},
272 {Token::SYMBOL_11, sym_3_poi_power}};
273 std::sort(symbol_powers.begin(), symbol_powers.end(),
274 [](const auto& a, const auto& b) { return a.second > b.second; });

276#ifdef K2_DEBUG_PRINT

285 }
286 std::cout << std::endl;

289 if (symbol_powers[0].second >
290 symbol_powers[1].second + MIN_AMPLITUDE_CUT_OFF_PARAMETER) {
291 return symbol_powers[0].first;
