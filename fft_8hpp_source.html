104 static void performFftFrequency(const std::span<const int16_t> input,
105 FftResults& results, uint32_t fixed_fft_bins = 0,
106 double min_fft_freq = 0.0,
107 double max_fft_freq = 20'000.0) {
108 // https://cplusplus.com/forum/beginner/251061/
109 // https://dsp.stackexchange.com/questions/19899/how-to-determine-snr-from-fft-of-measured-data
110 // https://download.ni.com/evaluation/pxi/Understanding%20FFTs%20and%20Windowing.pdf
111 results.reset();
112
113 const uint32_t num_samples = input.size();
114 const uint32_t fft_bins = fixed_fft_bins == 0 ? num_samples : fixed_fft_bins;
115
116 // The frequency resolution of the FFT (bin width)
117 const double delta_f = static_cast<double>(AUDIO_SAMPLE_RATE) / fft_bins;
118
119 // Load in and normalize the data
120 std::vector<double> in; // Data in the time domain
121 for (const auto sample : input) {
122 in.push_back(static_cast<double>(sample) / 32768.0);
123 }
124
125 // Output is n / 2 + 1 complex numbers per
126 // https://www.fftw.org/doc/One_002dDimensional-DFTs-of-Real-Data.html
127 std::vector<std::complex<double>> out(in.size() / 2 + 1);
128
129 // Perform the FFT
130 fftw_plan fft_plan = fftw_plan_dft_r2c_1d(
131 in.size(), in.data(), reinterpret_cast<fftw_complex*>(out.data()), FFTW_ESTIMATE);
132 fftw_execute(fft_plan);
133 fftw_destroy_plan(fft_plan);
134
135 double max_amplitude = 0.0;
136 double freq_at_max_amplitude = 0.0;
137 size_t max_amplitude_index = 0;
138 double average_amplitude = 0.0;
139 for (int K = 1; K < out.size(); ++K) {
140 const double frequency = K * delta_f;
141 if (frequency < min_fft_freq) {
142 continue; // Skip frequencies below 1kHz
143 }
144 if (frequency > max_fft_freq) {
145 break; // Stop at 4kHz
146 }
147
148 const double amplitude = std::abs(out[K]) * 2.0 / num_samples;
149
150 if (amplitude > max_amplitude) {
151 max_amplitude = amplitude;
152 freq_at_max_amplitude = frequency;
153 max_amplitude_index = K;
154 }
155
156 average_amplitude += amplitude;
157 results.frequency_amplitude.push_back({frequency, amplitude});
158 }
159 results.max_amplitude = {freq_at_max_amplitude, max_amplitude};
160 results.max_amplitude_index = max_amplitude_index;
161 results.average_amplitude = average_amplitude / out.size();
162
163 // Normalize the amplitudes to the maximum amplitude
164 for (auto& res : results.frequency_amplitude) {
165 results.normalized_frequency_amplitude.push_back(
166 {res.frequency, res.amplitude / max_amplitude});
167 // results.normalized_frequency_amplitude.push_back(
168 }
169
170 KTL_ASSERT(results.normalized_frequency_amplitude.size() ==
171 results.frequency_amplitude.size());
172
173 // Calculate the variance/standard deviation of the amplitudes and normalize
174 // the amplitudes
175 {
176 double normalized_variance = 0.0;
177 for (auto& res : results.normalized_frequency_amplitude) {
178 normalized_variance += std::pow(res.amplitude - results.average_amplitude, 2);
179 }
180 normalized_variance /= results.normalized_frequency_amplitude.size();
181
182 results.normalized_amplitude_variance = normalized_variance;
183 results.normalized_amplitude_std_dev = std::sqrt(normalized_variance);
184 }
185 }
