178 Token detectSymbolFromFftResults(const Fft::FftResults& results) const {
179 std::vector<Fft::FftResults::FreqAmp> sorted_peaks = results.getNormalizedPeaks(
180 NUM_PEAK_FREQS_TO_CHECK, MIN_AMPLITUDE_CUT_OFF_PARAMETER, MIN_FREQ_CUT_OFF,
181 MAX_FREQ_CUT_OFF);
182
183 // std::cout << "f,A: ";
184 // for (const auto& peak : sorted_peaks) {
185 // std::cout << "[" << peak.frequency << ", " << peak.amplitude << "]";
186 // }
187 // std::cout << std::endl;
188
189 // size_t num_0_peaks_of_interest = 0;
190 // size_t num_1_peaks_of_interest = 0;
191 double sym_0_poi_power = 0;
192 double sym_1_poi_power = 0;
193
194 // Start at a non-zero for the sake of a sensible SNR value later on.
195 // SNR would be better calculated by taking the 'integral' (summation, discrete
196 // probability here) of the whole FFT and then subtracting the peaks of interest.
197 double power_outside_poi = 0.2;
198
199 for (const auto& res : sorted_peaks) {
200 const double peak_to_check = res.frequency;
201 bool peak_used_by_symbol = false;
202
203 for (double peak_of_interest : SYM_A_00_PEAKS_OF_INTEREST) {
204 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
205 sym_0_poi_power += res.amplitude;
206 peak_used_by_symbol = true;
207 }
208 }
209 for (double peak_of_interest : SYM_D_01_PEAKS_OF_INTEREST) {
210 if (std::abs(peak_to_check - peak_of_interest) < PEAK_OF_INTEREST_ENVELOPE_HZ) {
211 sym_1_poi_power += res.amplitude;
212 peak_used_by_symbol = true;
213 }
214 }
215
216 if (!peak_used_by_symbol) {
217 power_outside_poi += res.amplitude;
218 }
219 }
220 sym_0_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
221 sym_1_poi_power /= NUM_PEAK_FREQS_TO_CHECK;
222 power_outside_poi /= NUM_PEAK_FREQS_TO_CHECK;
223
224 const double average_working_power =
225 (sym_0_poi_power + sym_1_poi_power) / NUM_PEAK_FREQS_TO_CHECK;
226
227 const double signal_to_noise_ratio = average_working_power / power_outside_poi;
228
229 if (average_working_power < MIN_AMPLITUDE_CUT_OFF_PARAMETER) {
230 return Token::UNKNOWN;
231 }
232
233 std::vector<std::pair<Token, double>> symbol_powers{
234 {Token::SYMBOL_0, sym_0_poi_power},
235 {Token::SYMBOL_1, sym_1_poi_power}};
236 std::sort(symbol_powers.begin(), symbol_powers.end(),
237 [](const auto& a, const auto& b) { return a.second > b.second; });
238
239#ifdef K3_DEBUG_PRINT
240 std::cout << "snr: " << signal_to_noise_ratio << " - ";
241 for (const auto& sym : symbol_powers) {
242 std::cout << (sym.first == Token::SYMBOL_0 ? "00"
243 : sym.first == Token::SYMBOL_1 ? "01"
244 : "XX")
245 << ":" << sym.second << ", ";
246 }
247 std::cout << std::endl;
248#endif
249
250 if (symbol_powers[0].second >
251 symbol_powers[1].second + MIN_AMPLITUDE_CUT_OFF_PARAMETER) {
252 return symbol_powers[0].first;
253 } else {
254 return Token::UNKNOWN;
255 }
256 }
